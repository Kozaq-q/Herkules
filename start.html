Slot #${slot.id}
`; }); html += `
+
Nowe zawody
`; grid.innerHTML = html; } // ═══════════════════════════════════════════════════ // ACTIONS // ═══════════════════════════════════════════════════ let selectedType = 'czworboj'; function selectType(t) { selectedType = t; document.getElementById('type-czworboj').classList.toggle('selected', t === 'czworboj'); document.getElementById('type-osf').classList.toggle('selected', t === 'osf'); } function openModal() { document.getElementById('new-city').value = ''; document.getElementById('new-name').value = ''; document.getElementById('new-date').valueAsDate = new Date(); selectType('czworboj'); document.getElementById('modal-new').classList.add('open'); setTimeout(() => document.getElementById('new-city').focus(), 50); } function closeModal() { document.getElementById('modal-new').classList.remove('open'); } function createSlot() { const city = document.getElementById('new-city').value.trim(); const name = document.getElementById('new-name').value.trim(); const date = document.getElementById('new-date').value; if (!city) { document.getElementById('new-city').focus(); return; } const ids = getSlotIds(); const newId = ids.length > 0 ?
